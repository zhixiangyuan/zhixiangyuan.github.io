# -l 参数表示 nc 将监听某个端口
# -p 表示监听的端口号为 $port
$> nc -l -p <$port>

# 与 [$ip]:[$port] 的服务器建立 tcp 连接
$> nc [$ip] [$port]

# 查看远程端口是否打开
# 其中 -z 参数表示不发送任何数据包，
# tcp 三次握手完后自动退出进程。有
# 了 -v 参数则会输出更多详细信息（verbose）
$> nc -zv [$ip] [$port]
