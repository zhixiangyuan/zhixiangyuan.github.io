class Solution {
    public int removeDuplicates(int[] nums) {
        int oldIndex = 1;
        int newIndex = 1;
        
        for (; oldIndex < nums.length; oldIndex++) {
            if (nums[oldIndex - 1] != nums[oldIndex]) {
                nums[newIndex] = nums[oldIndex];
                newIndex++;
            }
        }
        return newIndex;
    }
}
